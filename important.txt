# frontend
- create component
ng generate component [name] --inline-templete

- create service
ng generate service [name]

- run the app
ng serve -o

# backend
- to run:
sudo mvn spring-boot:run

- build & clean
sudo mvn clean install

- build and run
sudo mvn clean install && sudo mvn spring-boot:run

- build images for all microservices 
sudo mvn clean package jib:dockerBuild
sudo mvn clean install -DskipTests

- stop container
sudo docker stop [name]

- remove container
sudo docker rm [name]

- container list
sudo docker ps -a

- images list
sudo docker images

- check logs
sudo docker logs [name]
